procedure testFollowFollowS {
    print a;                
    read b;                 
    c = a + b + d;          

    while (		a 	== 	b	) {        
        c = a + b + d;      
        read b;             
        print a;            
        while (c == d) {    
        	a = b + c + d;  
            b = a 	+ c + d;  
            c = a + b + d;  
        }
        read a;             
        read b;             
    }

    print c;                
    print d;                
    if (a 	!= 	b		) then {      
        d = a + b + c;      
        read a;             
        print b;            
    } else {
        d = a + b + c;      
        read c;             
        print b;            
    }

    a = b + c + d;          
    read a;                 
    print a;                
}
